import { Meta, Canvas, Story, ArgsTable } from '@storybook/addon-docs'
import { aetherStoryDocs } from 'aetherspace/schemas'

import BackIcon from './BackIcon'
import HomeIcon from './HomeIcon'

import ReactIcon from './ReactIcon'
import ExpoIcon from './ExpoIcon'

<Meta title="App / Icons" />

export const createDocs = (Component) => (args) => {
    return <Component {...args} />
}

# App Icons

App icons can be found at `apps/app/icons/`

Thanks to yarn workspaces can also import them from other packages or apps via e.g.

`import { HomeIcon, BackIcon } from 'app/icons'`

## HomeIcon

export const homeIconDocs = createDocs(HomeIcon)
export const homeIconDocConfig = aetherStoryDocs({ HomeIcon })

<Canvas>
    <Story
        name="HomeIcon"
        args={homeIconDocConfig.args}
        argTypes={homeIconDocConfig.argTypes}
    >
        {homeIconDocs.bind({})}
    </Story>
</Canvas>

<ArgsTable story="HomeIcon" />
