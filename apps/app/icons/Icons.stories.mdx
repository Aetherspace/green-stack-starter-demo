    return <Component {...args} />

# App Icons

App icons can be found at `apps/app/icons/`

Thanks to yarn workspaces can also import them from other packages or apps via

`import { HomeIcon } from 'app/icons'`

## HomeIcon

export const HomeIconDocs = createDocs(HomeIcon)

<Canvas>
    <Story
        name="HomeIcon"
        args={{ fill: "#000000" }}
        argTypes={{
            fill: { control: { type: 'color' } }
        }}
    >
        {HomeIconDocs.bind({})}
    </Story>
</Canvas>
